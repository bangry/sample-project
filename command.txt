git branch <branch-name>     #브랜치 만들기
git branch                   #로컬 브랜치 목록 조회
git branch  -r               #서버 브랜치 목록 조회
git branch  -a               #로컬/서버 브랜치 목록 조회
git checkout <branch-name>   # 브랜치 전환
